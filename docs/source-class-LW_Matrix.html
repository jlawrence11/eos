 78:     /**
 79:      * Create a matrix based on string input similar to the TI Calculators
 80:      * input string "[1,2,3;4,5,6;7,8,9]" is the equivalent of the matrix:
 81:      * <pre>
 82:      * | 1  2  3 |
 83:      * | 4  5  6 |
 84:      * | 7  8  9 |
 85:      * </pre>
 86:      *
 87:      * @param String $mText The matrix in string format to assign to the current object
 88:      * @return Boolean True if is passes verification after being converted
 89:      * @throws Exception If the input text is not in a valid format
 90:      */
 91:     public function _assign($mText)
 92:     {
 93:         if(trim($mText)=="")
 94:             return false;
 95:         
 96:         $mText = preg_replace("/\s/", "", $mText);
 97:         if(!preg_match("/^\[(([\-]*[0-9\. ]+[,]{0,1})+[;]{0,1})*\]$/", $mText)) {
 98:             throw new Exception("'{$mText}' is not a valid input", LW_MATRIX_E_INVALID_INPUT);
 99:             return false;
100:         }
101:         $mText = preg_replace("/(\[|\])/", "", $mText);
102:         
103:         $rows = explode(";", $mText);
104:         $i=0;$j=0;
105:         foreach($rows as $row)
106:         {
107:             $cols = explode(",", $row);
108:             foreach($cols as $value)
109:             {
110:                 $this->matrix[$i][$j] = $value;
111:                 $j++;
112:             }
113:             $i++;
114:             $j = 0;
115:         }
116:         
117:         return $this->_verify();
118:     }

438:     /**
439:      * Get the Matrix Determinant
440:      * 
441:      * Finds the determinant of the square matrix, user should not use
442:      * the parameter, as that is meant to allow recursive calling
443:      * of this function from within itself.
444:      *
445:      * @link http://en.wikipedia.org/wiki/Matrix_determinant
446:      * @param Array $mArray The array to find a determinate of
447:      * @return Float The Determinate of the square matrix
448:      * @throws Exception If matrix is 1,1 or is not square
449:      */
450:     public function getDeterminant($mArray=false)
451:     {
452:         if(!$mArray) $mArray = $this->matrix;
453:         //print_r($mArray);
454:         if(!$this->isSquare($mArray)) {
455:             $m = $this->toString($mArray);
456:             throw new Exception("'{$m}' is not a square matrix", LW_MATRIX_E_NOT_SQUARE);
457:             return false;
458:         }
459:         $n = $this->_getN($mArray);
460:         if($n < 1){
461:             throw new Exception("1 by 1 Matrix cannot have a determinant", LW_MATRIX_E_NO_MATRIX);
462:             return 0; // how did it get this far?
463:         } elseif ($n == 1) {
464:             $det = $mArray[0][0];
465:         } elseif ($n == 2) {
466:             $det = $mArray[0][0]*$mArray[1][1] - $mArray[1][0]*$mArray[0][1];
467:         } else {
468:             $det = 0;
469:             $nArray = array();
470:             for($j1=0;$j1<$n;$j1++) {
471:                 for($i=1;$i<$n;$i++) {
472:                     $j2 = 0;
473:                     for($j=0;$j<$n;$j++) {
474:                         if($j==$j1) {
475:                             continue;
476:                         }
477:                         $nArray[$i-1][$j2] = $mArray[$i][$j];
478:                         $j2++;
479:                     }
480:                 }
481:                 $det += pow(-1,2+$j1)*$mArray[0][$j1]*$this->getDeterminant($nArray);
482:             }
483:         }
484:         return $det;
485:     }
