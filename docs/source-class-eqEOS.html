 62:  *
 63:  * An EOS that can safely parse equations from unknown sources returning
 64:  * the calculated value of it.  Can also handle solving equations with
 65:  * variables, if the variables are defined (useful for the Graph creation
 66:  * that the second and extended class in this file provides. {@see eqGraph})
 67:  * This class was created for PHP4 in 2005, updated to fully PHP5 in 2013.
 68:  * 
 69:  * @author Jon Lawrence <jlawrence11@gmail.com>

 78:      *Private variables
 79:      */
 80:     private $postFix;
 81:     private $inFix;
 82:     /**#@-*/
 83:     /**#@+
 84:      * Protected variables
 85:      */
 86:     //What are opening and closing selectors
 87:     protected $SEP = array('open' => array('(', '['), 'close' => array(')', ']'));
 88:     //Top presedence following operator - not in use
 89:     protected $SGL = array('!');
 90:     //Order of operations arrays follow
 91:     protected $ST = array('^');
 92:     protected $ST1 = array('/', '*', '%');
 93:     protected $ST2 = array('+', '-');
 94:     //Allowed functions
 95:     protected $FNC = array('sin', 'cos', 'tan', 'csc', 'sec', 'cot');

 97:     /**
 98:      * Construct method
 99:      *
100:      * Will initiate the class.  If variable given, will assign to
101:      * internal variable to solve with this::solveIF() without needing
102:      * additional input.  Initializing with a variable is not suggested.
103:      *
104:      * @see eqEOS::solveIF()
105:      * @param String $inFix Standard format equation
106:      */
107:     public function __construct($inFix = null) {
108:         $this->inFix = (isset($inFix)) ? $inFix : null;

112:     /**
113:      * Check Infix for opening closing pair matches.
114:      *
115:      * This function is meant to solely check to make sure every opening
116:      * statement has a matching closing one, and throws an exception if
117:      * it doesn't.
118:      *
119:      * @param String $infix Equation to check
120:      * @throws Exception if malformed.
121:      * @return Bool true if passes - throws an exception if not.
122:      */
123:     private function checkInfix($infix) {
124:         if(trim($infix) == "") {
125:             throw new Exception("No Equation given", EQEOS_E_NO_EQ);
126:             return false;
127:         }
128:         //Make sure we have the same number of '(' as we do ')'
129:         // and the same # of '[' as we do ']'
130:         if(substr_count($infix, '(') != substr_count($infix, ')')) {
131:             throw new Exception("Mismatched parenthesis in '{$infix}'", EQEOS_E_NO_SET);
132:             return false;
133:         } elseif(substr_count($infix, '[') != substr_count($infix, ']')) {
134:             throw new Exception("Mismatched brackets in '{$infix}'", EQEOS_E_NO_SET);
135:             return false;
136:         }
137:         $this->inFix = $infix;
138:         return true;
139:     }

141:     /**
142:      * Infix to Postfix
143:      *
144:      * Converts an infix (standard) equation to postfix (RPN) notation.
145:      * Sets the internal variable $this->postFix for the eqEOS::solvePF()
146:      * function to use.
147:      *
148:      * @link http://en.wikipedia.org/wiki/Infix_notation Infix Notation
149:      * @link http://en.wikipedia.org/wiki/Reverse_Polish_notation Reverse Polish Notation
150:      * @param String $infix A standard notation equation
151:      * @return Array Fully formed RPN Stack
152:      */
153:     public function in2post($infix = null) {
154:         // if an equation was not passed, use the one that was passed in the constructor
155:         $infix = (isset($infix)) ? $infix : $this->inFix;
156:         
157:         //check to make sure 'valid' equation
158:         $this->checkInfix($infix);
159:         $pf = array();
160:         $ops = new phpStack();
161:         $vars = new phpStack();
162: 
163:         // remove all white-space
164:         preg_replace("/\s/", "", $infix);
165: 
166:         // Create postfix array index
167:         $pfIndex = 0;
168: 
169:         //what was the last character? (useful for decerning between a sign for negation and subtraction)
170:         $lChar = '';
171: 
172:         //loop through all the characters and start doing stuff ^^
173:         for($i=0;$i<strlen($infix);$i++) {
174:             // pull out 1 character from the string
175:             $chr = substr($infix, $i, 1);
176:             
177:             // if the character is numerical
178:             if(preg_match('/[0-9.]/i', $chr)) {
179:                 // if the previous character was not a '-' or a number
180:                 if((!preg_match('/[0-9.]/i', $lChar) && ($lChar != "")) && ($pf[$pfIndex]!="-"))
181:                     $pfIndex++; // increase the index so as not to overlap anything
182:                 // Add the number character to the array
183:                 $pf[$pfIndex] .= $chr;
184:             }
185:             // If the character opens a set e.g. '(' or '['
186:             elseif(in_array($chr, $this->SEP['open'])) {
187:                 // if the last character was a number, place an assumed '*' on the stack
188:                 if(preg_match('/[0-9.]/i', $lChar))
189:                     $ops->push('*');
190: 
191:                 $ops->push($chr);
192:             }
193:             // if the character closes a set e.g. ')' or ']'
194:             elseif(in_array($chr, $this->SEP['close'])) {
195:                 // find what set it was i.e. matches ')' with '(' or ']' with '['
196:                 $key = array_search($chr, $this->SEP['close']);
197:                 // while the operator on the stack isn't the matching pair...pop it off
198:                 while($ops->peek() != $this->SEP['open'][$key]) {
199:                     $nchr = $ops->pop();
200:                     if($nchr)
201:                         $pf[++$pfIndex] = $nchr;
202:                     else {
203:                         throw new Exception("Error while searching for '". $this->SEP['open'][$key] ."' in '{$infix}'.", EQEOS_E_NO_SET);
204:                         return false;
205:                     }
206:                 }
207:                 $ops->pop();
208:             }
209:             // If a special operator that has precedence over everything else
210:             elseif(in_array($chr, $this->ST)) {
211:                 $ops->push($chr);
212:                 $pfIndex++;
213:             }
214:             // Any other operator other than '+' and '-'
215:             elseif(in_array($chr, $this->ST1)) {
216:                 while(in_array($ops->peek(), $this->ST1) || in_array($ops->peek(), $this->ST))
217:                     $pf[++$pfIndex] = $ops->pop();
218: 
219:                 $ops->push($chr);
220:                 $pfIndex++;
221:             }
222:             // if a '+' or '-'
223:             elseif(in_array($chr, $this->ST2)) {
224:                 // if it is a '-' and the character before it was an operator or nothingness (e.g. it negates a number)
225:                 if((in_array($lChar, array_merge($this->ST1, $this->ST2, $this->ST)) || $lChar=="") && $chr=="-") {
226:                     // increase the index because there is no reason that it shouldn't..
227:                     $pfIndex++;
228:                     $pf[$pfIndex] .= $chr; 
229:                 }
230:                 // Otherwise it will function like a normal operator
231:                 else {
232:                     while(in_array($ops->peek(), array_merge($this->ST1, $this->ST2, $this->ST)))
233:                         $pf[++$pfIndex] = $ops->pop();
234:                     $ops->push($chr);
235:                     $pfIndex++;
236:                 }
237:             }
238:             // make sure we record this character to be refered to by the next one
239:             $lChar = $chr;
240:         }
241:         // if there is anything on the stack after we are done...add it to the back of the RPN array
242:         while(($tmp = $ops->pop()) !== false)
243:             $pf[++$pfIndex] = $tmp;
244: 
245:         // re-index the array at 0
246:         $pf = array_values($pf);
247:         
248:         // set the private variable for later use if needed
249:         $this->postFix = $pf;
250: 
251:         // return the RPN array in case developer wants to use it fro some insane reason (bug testing ;]
252:         return $pf;
253:     } //end function in2post

255:     /**
256:      * Solve Postfix (RPN)
257:      * 
258:      * This function will solve a RPN array. Default action is to solve
259:      * the RPN array stored in the class from eqEOS::in2post(), can take
260:      * an array input to solve as well, though default action is prefered.
261:      *
262:      * @link http://en.wikipedia.org/wiki/Reverse_Polish_notation Postix Notation
263:      * @param Array $pfArray RPN formatted array. Optional.
264:          * @return Float Result of the operation.
265:      */
266:     public function solvePF($pfArray = null) {
267:         // if no RPN array is passed - use the one stored in the private var
268:         $pf = (!is_array($pfArray)) ? $this->postFix : $pfArray;
269:         
270:         // create our temporary function variables
271:         $temp = array();
272:         $tot = 0;
273:         $hold = 0;
274: 
275:         // Loop through each number/operator 
276:         for($i=0;$i<count($pf); $i++) {
277:             // If the string isn't an operator, add it to the temp var as a holding place
278:             if(!in_array($pf[$i], array_merge($this->ST, $this->ST1, $this->ST2))) {
279:                 $temp[$hold++] = $pf[$i];
280:             }
281:             // ...Otherwise perform the operator on the last two numbers 
282:             else {
283:                 switch ($pf[$i]) {
284:                     case '+':
285:                         $temp[$hold-2] = $temp[$hold-2] + $temp[$hold-1];
286:                         break;
287:                     case '-':
288:                         $temp[$hold-2] = $temp[$hold-2] - $temp[$hold-1];
289:                         break;
290:                     case '*':
291:                         $temp[$hold-2] = $temp[$hold-2] * $temp[$hold-1];
292:                         break;
293:                     case '/':
294:                         if($temp[$hold-1] == 0) {
295:                             throw new Exception("Division by 0 on: '{$temp[$hold-2]} / {$temp[$hold-1]}' in {$this->inFix}", EQEOS_E_DIV_ZERO);
296:                             return false;
297:                         }
298:                         $temp[$hold-2] = $temp[$hold-2] / $temp[$hold-1];
299:                         break;
300:                     case '^':
301:                         $temp[$hold-2] = pow($temp[$hold-2], $temp[$hold-1]);
302:                         break;
303:                     case '%':
304:                         if($temp[$hold-1] == 0) {
305:                             throw new Exception("Division by 0 on: '{$temp[$hold-2]} % {$temp[$hold-1]}' in {$this->inFix}", EQEOS_E_DIV_ZERO);
306:                             return false;
307:                         }
308:                         $temp[$hold-2] = bcmod($temp[$hold-2], $temp[$hold-1]);
309:                         break;
310:                 }
311:                 // Decrease the hold var to one above where the last number is 
312:                 $hold = $hold-1;
313:             }
314:         }
315:         // return the last number in the array 
316:         return $temp[$hold-1];
317: 
318:     } //end function solvePF

321:     /**
322:      * Solve Infix (Standard) Notation Equation
323:      *
324:      * Will take a standard equation with optional variables and solve it. Variables
325:      * must begin with '&' will expand to allow variables to begin with '$' (TODO)
326:      * The variable array must be in the format of 'variable' => value. If
327:      * variable array is scalar (ie 5), all variables will be replaced with it.
328:      *
329:      * @param String $infix Standard Equation to solve
330:      * @param String|Array $vArray Variable replacement
331:      * @return Float Solved equation
332:      */
333:     function solveIF($infix, $vArray = null) {
334:         $infix = ($infix != "") ? $infix : $this->inFix;
335:         
336:         //Check to make sure a 'valid' expression
337:         $this->checkInfix($infix);
338: 
339:         $ops = new phpStack();
340:         $vars = new phpStack();
341: 
342:         //remove all white-space
343:         preg_replace("/\s/", "", $infix);
344:         if(DEBUG)
345:             $hand=fopen("eq.txt","a");
346: 
347:         //Find all the variables that were passed and replaces them
348:         while((preg_match('/(.){0,1}[&$]([a-zA-Z]+)(.){0,1}/', $infix, $match)) != 0) {
349: 
350:             if(DEBUG)
351:                 fwrite($hand, "{$match[1]} || {$match[3]}\n");
352:             // Ensure that the variable has an operator or something of that sort in front and back - if it doesn't, add an implied '*'
353:             if((!in_array($match[1], array_merge($this->ST, $this->ST1, $this->ST2, $this->SEP['open'])) && $match[1] != "") || is_numeric($match[1])) //$this->SEP['close'] removed
354:                 $front = "*";
355:             else
356:                 $front = "";
357: 
358:             if((!in_array($match[3], array_merge($this->ST, $this->ST1, $this->ST2, $this->SEP['close'])) && $match[3] != "") || is_numeric($match[3])) //$this->SEP['open'] removed
359:                 $back = "*";
360:             else
361:                 $back = "";
362:             
363:             //Make sure that the variable does have a replacement
364:             if(!isset($vArray[$match[2]]) && (!is_array($vArray != "") && !is_numeric($vArray))) {
365:                 throw new Exception("Variable replacement does not exist for '". substr($match[0], 1, -1) ."' in {$this->inFix}", EQEOS_E_NO_VAR);
366:                 return false;
367:             } elseif(!isset($vArray[$match[2]]) && (!is_array($vArray != "") && is_numeric($vArray))) {
368:                 $infix = str_replace($match[0], $match[1] . $front. $vArray. $back . $match[3], $infix);
369:             } elseif(isset($vArray[$match[2]])) {
370:                 $infix = str_replace($match[0], $match[1] . $front. $vArray[$match[2]]. $back . $match[3], $infix);
371:             }
372:         }
373: 
374:         if(DEBUG)
375:             fwrite($hand, "$infix\n");
376: 
377:         // Finds all the 'functions' within the equation and calculates them 
378:         // NOTE - when using function, only 1 set of paranthesis will be found, instead use brackets for sets within functions!! 
379:         while((preg_match("/(". implode("|", $this->FNC) . ")\(([^\)\(]*(\([^\)]*\)[^\(\)]*)*[^\)\(]*)\)/", $infix, $match)) != 0) {
380:             $func = $this->solveIF($match[2]);
381:             switch($match[1]) {
382:                 case "cos":
383:                     $ans = cos($func);
384:                     break;
385:                 case "sin":
386:                     $ans = sin($func);
387:                     break;
388:                 case "tan":
389:                     $ans = tan($func);
390:                     break;
391:                 case "sec":
392:                     $tmp = cos($func);
393:                     if($tmp == 0) {
394:                         throw new Exception("Division by 0 on: 'sec({$func}) = 1/cos({$func})' in {$this->inFix}", EQEOS_E_DIV_ZERO);
395:                         return false;
396:                     }
397:                     $ans = 1/$tmp;
398:                     break;
399:                 case "csc":
400:                     $tmp = sin($func);
401:                     if($tmp == 0) {
402:                         throw new Exception("Division by 0 on: 'csc({$func}) = 1/sin({$func})' in {$this->inFix}", EQEOS_E_DIV_ZERO);
403:                         return false;
404:                     }
405:                     $ans = 1/$tmp;
406:                     break;
407:                 case "cot":
408:                     $tmp = tan($func);
409:                     if($tmp == 0) {
410:                         throw new Exception("Division by 0 on: 'cot({$func}) = 1/tan({$func})' in {$this->inFix}", EQEOS_E_DIV_ZERO);
411:                         return false;
412:                     }
413:                     $ans = 1/$tmp;
414:                     break;
415:                 default:
416:                     break;
417:             }
418:             $infix = str_replace($match[0], $ans, $infix);
419:         }
420:         if(DEBUG)
421:             fclose($hand);
422:         return $this->solvePF($this->in2post($infix));
423: 
424: 
425:     } //end function solveIF
